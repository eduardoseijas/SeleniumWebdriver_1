@echo off
setlocal

REM === 1) Ir al proyecto en D ===
cd /d D:\Practica_Selenium_2 || (echo [ERROR] No existe D:\Practica_Selenium_2 & exit /b 1)

REM === 2) Python absoluto (AJUSTA si tu Python est√° en otra ruta) ===
set "PYTHON_EXE=C:\Users\Usuario\AppData\Local\Programs\Python\Python313\python.exe"
if not exist "%PYTHON_EXE%" set "PYTHON_EXE=C:\Program Files\Python313\python.exe"
if not exist "%PYTHON_EXE%" (
  echo [ERROR] No encuentro python.exe. Cambia PYTHON_EXE a tu ruta real.
  exit /b 1
)

REM === 3) Crear/activar venv ===
if not exist ".venv\Scripts\activate.bat" (
  "%PYTHON_EXE%" -m venv .venv || (echo [ERROR] No se pudo crear venv & exit /b 1)
)
call ".venv\Scripts\activate.bat" || (echo [ERROR] No se pudo activar venv & exit /b 1)

REM === 4) Instalar deps usando el python de la venv (sin 'pip' directo) ===
".venv\Scripts\python.exe" -m pip install -U pip
if exist requirements.txt (
  ".venv\Scripts\python.exe" -m pip install -r requirements.txt
) else (
  ".venv\Scripts\python.exe" -m pip install pytest selenium
)

REM === 5) Ejecutar tests (archivo o todos) ===
REM Archivo concreto:
".venv\Scripts\python.exe" -m pytest -s -v "25_Fixture\Test_Pytest_Fixture_2.py"

REM Para todos, usa en su lugar:
REM ".venv\Scripts\python.exe" -m pytest -s -v

exit /b %ERRORLEVEL%
